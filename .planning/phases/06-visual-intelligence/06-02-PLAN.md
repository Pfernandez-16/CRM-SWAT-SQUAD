    **2a — Add dealsVelocity computed.**
    After the `dealsLossRows` computed block (around line 323), add:
    ```javascript
    const dealsVelocity = computed(function() {
        if (!reportData.value || !reportData.value.dealsReport) return null;
        const dr = reportData.value.dealsReport;
        const cur  = dr.avgDaysToClose     || 0;
        const prev = dr.prevAvgDaysToClose  || 0;
        let delta;
        if (prev === 0 && cur === 0) { delta = '0.0'; }
        else if (prev === 0 && cur > 0) { delta = '100.0'; }
        else { delta = ((cur - prev) / prev * 100).toFixed(1); }
        return { current: cur, previous: prev, delta: delta };
    });
    ```

    **2b — Add chart instance ref and watch(embudoRows) to draw/redraw the funnel chart.**
    After the dealsVelocity computed (still inside the Vue setup function), add the chart ref and watcher:

    ```javascript
    // Chart.js funnel chart for Embudo General (CHART-01)
    let embudoChartInstance = null;

    watch(embudoRows, function(rows) {
        // Only render count rows (skip amount rows)
        const countRows = rows.filter(function(r) { return r.type === 'count'; });
        const labels = countRows.map(function(r) { return r.label; });
        const counts = countRows.map(function(r) {
            return (r.metric && r.metric.total && r.metric.total.count) || 0;
        });

        const canvas = document.getElementById('embudoChart');
        if (!canvas) return;

        // Destroy prior instance to avoid canvas re-use error
        if (embudoChartInstance) {
            embudoChartInstance.destroy();
            embudoChartInstance = null;
        }

        // Bar widths proportional to count create funnel illusion
        embudoChartInstance = new Chart(canvas, {
            type: 'bar',
            data: {
                labels: labels,
                datasets: [{
                    label: 'Leads',
                    data: counts,
                    backgroundColor: 'rgba(99, 149, 255, 0.75)',
                    borderColor: 'rgba(99, 149, 255, 1)',
                    borderWidth: 1,
                    borderRadius: 3
                }]
            },
            options: {
                indexAxis: 'y',
                responsive: true,
                maintainAspectRatio: false,
                plugins: {
                    legend: { display: false },
                    tooltip: {
                        callbacks: {
                            label: function(ctx) { return ' ' + ctx.parsed.x + ' leads'; }
                        }
                    }
                },
                scales: {
                    x: {
                        beginAtZero: true,
                        grid: { color: 'rgba(255,255,255,0.06)' },
                        ticks: { color: '#aaa' }
                    },
                    y: {
                        grid: { display: false },
                        ticks: { color: '#ccc', font: { size: 11 } }
                    }
                }
            }
        });
    }, { deep: false });
    ```
