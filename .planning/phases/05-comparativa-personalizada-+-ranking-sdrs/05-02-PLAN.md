Change:
```javascript

});
```
to:
```javascript
watch(compareType, function (newType) {
    if (newType === 'custom') {
        nextTick(function () {
            if (customPrevFlatpickrInstance) {
                customPrevFlatpickrInstance.destroy();
                customPrevFlatpickrInstance = null;
            }
            customPrevFlatpickrInstance = flatpickr('.custom-prev-picker', {
                mode: 'range',
                dateFormat: 'Y-m-d',
                locale: { firstDayOfWeek: 1, rangeSeparator: ' a ' },
                onChange: function (selectedDates) {
                    if (selectedDates.length === 2) {
                        customPrevRange.value = { start: selectedDates[0], end: selectedDates[1] };
                    }
                }
            });
        });
    } else {
        if (customPrevFlatpickrInstance) {
            customPrevFlatpickrInstance.destroy();
            customPrevFlatpickrInstance = null;
        }
        generateReport();
    }
});
```
Note: when newType === 'custom' we do NOT auto-call generateReport() — user must pick dates first then click "Generar Reporte" manually.

Style guide: App.html uses const/let freely (Vue 3 CDN, modern browser). The only ES5 constraint is Analytics.js (GAS backend).
