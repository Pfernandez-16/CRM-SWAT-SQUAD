    saving.value = true;
    var lead = handoffLead.value;
    var payload = {
        leadId: lead._id || '',
        leadRow: lead._row || '',
        aeEmail: handoffData.aeEmail || '',
        notas: handoffData.notas_bandeja_plata || '',

        }
    };
    google.script.run

            saving.value = false;
            if (result && result.status === 'success') {
                showToast('🎉 ' + result.message, 'success');
                showHandoffModal.value = false;
                showLeadModal.value = false;

                showToast('❌ ' + (result.message || 'Error en el handoff'), 'error');

        })
        .withFailureHandler(function (err) {

}
