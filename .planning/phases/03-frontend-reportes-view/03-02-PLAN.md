4. **Helper: getSegmentValue** — extracts count or amount from a segment metric:
```javascript
function getSegmentValue(metric, segment) {
  // metric = { total: {count, pct, delta}, manufacturers: {...}, individuals: {...} }
  // OR { total: {amount, delta}, ... } for amount-type
  if (!metric || !metric[segment]) return { count: 0, pct: '0.0', delta: '0.0' };
  return metric[segment];
}

function formatAmount(val) {
  if (!val && val !== 0) return '$0';
  return '$' + Number(val).toLocaleString('en-US', { minimumFractionDigits: 0, maximumFractionDigits: 0 });
}
```
