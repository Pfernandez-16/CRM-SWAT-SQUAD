5. Add formatDateYMD helper:
```javascript
function formatDateYMD(date) {
  if (!date) return '';
  const year = date.getFullYear();
  const month = String(date.getMonth() + 1).padStart(2, '0');
  const day = String(date.getDate()).padStart(2, '0');
  return year + '-' + month + '-' + day;
}
```

6. Add generateReport, cancelReport functions:
```javascript
let reportAbortController = null;

function generateReport() {
  if (!dateRange.value.start || !dateRange.value.end) return;
  reportLoading.value = true;
  reportError.value = null;
  reportAbortController = new AbortController();

  const dateIn = formatDateYMD(dateRange.value.start);
  const dateOut = formatDateYMD(dateRange.value.end);

  google.script.run
    .withSuccessHandler(function(jsonString) {
      if (reportAbortController && reportAbortController.signal.aborted) return;
      try {
        const data = typeof jsonString === 'string' ? JSON.parse(jsonString) : jsonString;
        if (data.error) {
          reportError.value = data.error;
          reportData.value = null;
        } else if (data.metadata && data.metadata.totalLeads === 0) {
          reportData.value = { _empty: true };
        } else {
          reportData.value = data;
        }
      } catch (e) {
        reportError.value = 'Error al procesar la respuesta: ' + e.message;
        reportData.value = null;
      }
      reportLoading.value = false;
    })
    .withFailureHandler(function(error) {
      if (reportAbortController && reportAbortController.signal.aborted) return;
      reportError.value = error.message || 'Error desconocido al generar el reporte';
      reportData.value = null;
      reportLoading.value = false;
    })
    .getSDRReport(dateIn, dateOut);
}

function cancelReport() {
  if (reportAbortController) {
    reportAbortController.abort();
  }
  reportLoading.value = false;
}
```

```javascript
function toggleSection(sectionId) {
  const idx = openSections.value.indexOf(sectionId);
  if (idx > -1) {
    openSections.value.splice(idx, 1);
  } else {
    openSections.value.push(sectionId);
  }
}

function scrollToSection(sectionId) {
  if (!openSections.value.includes(sectionId)) {
    openSections.value.push(sectionId);
  }
  nextTick(() => {
    const el = document.getElementById('section-' + sectionId);
    if (el) el.scrollIntoView({ behavior: 'smooth', block: 'start' });
  });
}
```

```javascript
          const diffMs = selectedDates[1].getTime() - selectedDates[0].getTime();
```
